    3 {-|
    4 Module      : MapApi
    5 Description : Contains functions to access the Bing Map Api
    6 
    7 This module handles accessing the Bing Maps api to return Locations for a given Query.
    8 -}
    9 module MapApi (Location(..), getLocation) where
   10 
   11 import System.Environment -- to get environment variables
   12 import Data.Text -- to pack strings to pass into lib functions
   13 
   14 import Network.Wreq -- Main REST request lib
   15 import Control.Lens -- Lib used to access results from requests
   16 import Data.Aeson.Lens -- Json Access
   17 
   18 -- | Location describing a rest request result
   19 data Location = Location { name :: String, -- ^ Name of the location found
   20                         entityType :: String, -- ^ Entitiy Type of the location (e.g. River)
   21                         confidence :: String -- ^ how confident the Api was that this is the correct result (low, medium or high)
   22                         }
   23 
   24 -- Print a location result
   25 instance Show Location where
   26     show (Location name entityType confidence) = "{" ++ name ++ " (" ++ entityType ++ ", Confidence: " ++ confidence ++ ")}"
   27 
   28 
   29 -- | Main node of the rest request
   30 mapsUrl :: String
   31 mapsUrl = "http://dev.virtualearth.net/REST/v1/Locations"
   32 
   33 -- | Get a list of all matches for the given query string
   34 getLocation :: String -- ^ Query to be searched
   35             -> IO [Location] -- ^ List of all locations returned from the api. Can be empty
   36 getLocation query  = if query == ""
   37                         then return []
   38                         else
   39                             do
   40                                 apiKey <- getEnv "BING_API_KEY"
   41                                 let opts = defaults & param "query" .~ [pack query]
   42                                                     & param "key" .~ [pack apiKey]
   43                                                     & param "c" .~ [pack "de"]
   44                                                     & param "userRegion" .~ [pack "DE"]
   45                                                     & param "userLocation" .~ [pack "48.153737,11.552366"]
   46                                 response <- getWith opts mapsUrl
   47                                 let numResult = fromIntegral (Prelude.head (numberOfResources response))
   48                                     range =  [0..numResult-1]
   49                                 return (Prelude.map (readLocation response) range)
   50 
   51 -- | Read in all relevant info from the given json to a Location Data object
   52 readLocation response index = Location
   53                                 (unpack (readName response index))
   54                                 (unpack (readEntityType response index))
   55                                 (unpack (readConfidence response index))
   56 
   57 -- Access functions for Location-json fields
   58 -- These are not typed because the types rely on the Json implementation which might vary
   59 
   60 -- | Read the name of the nth Location from the response body
   61 readName response index = response ^. responseBody . key "resourceSets" . nth 0 . key "resources" . nth index . key "name" . _String
   62 
   63 -- | Read the entity ype of the nth Location from the response body
   64 readEntityType response index =  response ^. responseBody . key "resourceSets" . nth 0 . key "resources" . nth index . key "entityType" . _String
   65 
   66 -- | Read the confidence of the nth Location from the response body
   67 readConfidence response index =  response ^. responseBody . key "resourceSets" . nth 0 . key "resources" . nth index . key "confidence" . _String
   68 
   69 -- | Read the info field in which the number of given responses is stored
   70 numberOfResources response = response ^.. responseBody . key "resourceSets" . nth 0 . key "estimatedTotal"  . _Integer
