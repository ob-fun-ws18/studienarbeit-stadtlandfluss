   26 -- | Checks and scores a list of answers for the river type
   27 checkRiver :: [String] -- ^ List of all answers given
   28              -> IO [ (Int, String)] -- ^ Score and likelihood of every answer
   29 checkRiver answers =  mapM (evaluateAnswer "River" answers) answers

   41 -- | Evaluates how likely a given answer is to be correct and how many points it should be worth
   42 -- Uses the Rest API to get all Locations matching the Answer.
   43 evaluateAnswer :: String -- ^ The entity type of the answer CountryRegion | River | PopulatedPlace
   44                -> [String] -- ^ List of all Answers to determine score multiplier
   45                -> String -- ^ The given Answer which will be scored
   46                -> IO (Int, String) -- ^ Score and likelihood
   47 evaluateAnswer queryType allQueries query =
   48         do locations <- getLocation query
   49            return (scoreAnswer queryType locations query allQueries)
   50 
   51 -- | Score a answer based on given locations
   52 scoreAnswer :: String -- ^ Entity Type which the location should have
   53             -> [Location] -- ^ List of locations matching the query
   54             -> String -- ^ Query / Answer to be scored
   55             -> [String] -- ^ All Answers used for score multiplier
   56             -> (Int, String) -- ^ Score and likelihood
   57 scoreAnswer queryType locs query allQueries
   58     = let loc = bestMatch queryType locs
   59           conf = maybe "Invalid" confidence loc
   60           points = maybe 0 (const 5) loc
   61           bonusFactor = calcBonusFactor query allQueries
   62       in (points * bonusFactor, conf)
   63 
   64 -- | Filter a sorted List of locations for the best match in a given entity Type
   65 bestMatch :: String -- ^ Entity type which the result MUST match
   66           -> [Location] -- ^ List of all Locations
   67           -> Maybe Location -- ^ Best Match for given type, nothing if no result for given type was found
   68 bestMatch expectedType locations = listToMaybe (filter (isType expectedType) locations)
   69 
   70 -- | determine if Location type matches
   71 isType :: String -- ^ Name of the type expected
   72        -> Location -- ^ Location to be checked
   73        -> Bool -- ^ true if entityType is as expected
   74 isType expetected location = entityType location == expetected
   75 
   76 -- | Calculates a score multiplier for a given Answer
   77 -- 4: if only one non empty answer was given. (No one else answered in that category)
   78 -- 2: The answer is unique. (There where other answers, but not equal answers)
   79 -- 1: The answer was given multiple times.
   80 -- Note: Empty answers will get score multipliers too, but since they will receive 0 points the multiplier does not matter
   81 calcBonusFactor :: String -- ^ Answer for which the multiplier is calculated
   82                 -> [String] -- ^ List of all Answers (including the one being evaluated)
   83                 -> Int -- ^ Score multiplier (1 | 2 | 4)
   84 calcBonusFactor query allQueries | length (filter (== query) allQueries) > 1 = 1 -- Duplicate answer
   85                                  | length (filter (== "") allQueries) == length allQueries - 1 = 4 -- Only answer
   86                                  | otherwise = 2 -- Unique answer
